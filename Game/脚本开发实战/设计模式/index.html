    # 状态机实现（Python伪代码）
def task_drawing(thread_id):
    flow_state = "读取当前任务"
    task_data = TaskData()
    
    while True:
        if flow_state == "读取当前任务":
            # 优先尝试快捷栏读取
            if read_task_from_quickbar(thread_id, task_data):
                logger.info(f"任务: {task_data.monster_name}, {task_data.map_name}")
                flow_state = "飞行"
            else:
                # 快捷栏无任务，打开任务面板
                flow_state = "查找赏金任务"
        
        elif flow_state == "查找赏金任务":
            if check_task_panel_open(thread_id):
                # 锁定识别范围（关键优化！）
                category_rect = get_task_category_rect(thread_id)
                
                # 查找"赏金任务"图标
                if dm_find_pic(thread_id, category_rect, "task_reward*.bmp", 0.9):
                    # 检查是否在打图任务详情页
                    detail_rect = get_task_detail_rect(thread_id)
                    if dm_find_pic(thread_id, detail_rect, "task_dig*.bmp", 0.9):
                        flow_state = "读取任务详情"
                    else:
                        # 点击赏金任务
                        click_position(thread_id, reward_pos.x + 20, reward_pos.y + 5)
                        dm_double_click(thread_id)
                else:
                    # 当前未接任务
                    close_task_panel(thread_id)
                    return False
            else:
                open_task_panel(thread_id)
        
        elif flow_state == "读取任务详情":
            # OCR识别任务详情
            task_content = dm_ocr_read(thread_id, task_detail_rect)
            
            # 解析多种任务文本格式（容错设计）
            if "埋他的" in task_content:
                # 格式: "挖宝XX正在地图名的X.Y埋他的宝贝"
                task_data.monster_name = extract_text(task_content, "挖宝", "正在")
                task_data.map_name = extract_text(task_content, "正在", "的")
                task_data.monster_x = extract_coord(task_content, task_data.map_name + "的", ".")
                task_data.monster_y = extract_coord(task_content, ".", "埋他")
                task_data.is_boss = False
            
            elif "前往制服" in task_content:
                # 贼王格式
                task_data.monster_name = extract_text(task_content, "一", "最近")
                task_data.map_name = extract_text(task_content, "最近在", "出现")
                task_data.is_boss = False
            
            # 验证数据完整性
            if task_data.monster_name and task_data.map_name:
                close_task_panel(thread_id)
                flow_state = "飞行"
            else:
                # 数据不完整，重新读取
                close_task_panel(thread_id)
                flow_state = "查找赏金任务"
        
        elif flow_state == "飞行":
            # 使用飞行旗快速传送
            if use_fly_flag(thread_id, task_data.map_name):
                flow_state = "寻路到指定地图"
            else:
                # 飞行旗失败，使用常规寻路
                flow_state = "寻路到指定地图"
        
        elif flow_state == "寻路到指定地图":
            if task_data.map_name == "建邺城":
                map_goto_datang_guojing(thread_id)
                flow_state = "移动到指定点"
            # ... 其他地图处理
        
        elif flow_state == "移动到指定点":
            map_move_to_point(thread_id, task_data.monster_x, 
                            task_data.monster_y, accuracy=5)
            flow_state = "点击强盗"
        
        elif flow_state == "点击强盗":
            click_npc(thread_id, task_data.monster_x, task_data.monster_y)
            flow_state = "对话并战斗"
        
        elif flow_state == "对话并战斗":

                logger.info("任务完成")
                flow_state = "读取当前任务"  # 循环执行下一个任务
            else:
                logger.error("任务执行失败")
        
        time.sleep(0.5)

class TaskRouter:
    """任务路由中心"""
    
    def __init__(self, thread_id: int):
        self.thread_id = thread_id
        self.current_task_type = None
    
    def route_task(self):
        """
        根据当前任务类型，路由到对应的任务模块
        这是路由中心的核心功能！
        """
        # 1. 读取当前任务类型（从UI配置、快捷键或游戏内判断）
        task_type = self.get_task_type()
        
        logger.info(f"路由中心: 任务类型={task_type}")
        
        # 2. 根据任务类型分发到不同模块
        if task_type == "打图":
            self.handle_drawing_task()
        elif task_type == "强盗":
            self.handle_bandit_task()
        elif task_type == "贼王":
            self.handle_boss_task()
        elif task_type == "押镖":
            self.handle_escort_task()
        elif task_type == "抓鬼":
            self.handle_ghost_task()
        else:
            logger.warning(f"未知任务类型: {task_type}")
    
    def get_task_type(self) -> str:
        """
        获取当前应该执行的任务类型
        来源：UI配置、快捷键或游戏内任务判定
        """
        config = ConfigHelper()
        section = f"窗口_{self.thread_id}"
        
        # 从配置文件读取任务类型
        task_type = config.get(section, "任务类型", "打图")
        return task_type

from multiprocessing import Process, Queue, Manager
from utils.logger import setup_logger

logger = setup_logger("ProcessManager")

class ProcessManager:
    """进程管理器"""
    
    def __init__(self):
        self.processes = {}  # {thread_id: Process}
        self.command_queues = {}
        self.status_queues = {}
        self.log_queues = {}
        
        # 共享配置
        self.manager = Manager()
        self.shared_config = self.manager.dict()
        
        logger.info("进程管理器初始化完成")
    
    def start_process(self, thread_id: int):
        """启动工作进程"""
        if thread_id in self.processes:
            logger.warning(f"进程 {thread_id} 已存在")
            return
        
        # 创建队列
        command_queue = Queue()
        status_queue = Queue()
        log_queue = Queue()
        
        self.command_queues[thread_id] = command_queue
        self.status_queues[thread_id] = status_queue
        self.log_queues[thread_id] = log_queue
        
        # 启动进程
        process = Process(
            target=worker_process_main,
            args=(thread_id, command_queue, status_queue, log_queue)
        )
        process.start()
        self.processes[thread_id] = process
        
        logger.info(f"进程 {thread_id} 启动成功，PID: {process.pid}")
    
    def stop_process(self, thread_id: int):
        """停止工作进程"""
        if thread_id not in self.processes:
            return
        
        # 发送停止命令
        self.command_queues[thread_id].put({'command': 'stop'})
        
        # 等待进程结束
        process = self.processes[thread_id]
        process.join(timeout=5)
        
        if process.is_alive():
            process.terminate()
        
        del self.processes[thread_id]
        logger.info(f"进程 {thread_id} 已停止")
