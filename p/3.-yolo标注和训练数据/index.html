import cv2
import os

video = cv2.VideoCapture("./wendao.wmv")  
num = 0
save_step = 30  # 每隔 30 帧保存一张
save_dir = "./demo_images"

os.makedirs(save_dir, exist_ok=True)

while True:
    ret, frame = video.read()
    if not ret:
        break
    num += 1
    if num % save_step == 0:
        cv2.imwrite(f"{save_dir}/{num}.jpg", frame)

video.release()

1

dataset/
├── images/     # 放所有待标注图片

⚠️ 注意：路径不能包含中文或空格，否则 LabelImg 无法保存。

打开 LabelImg 并设置保存路径

  1. 点击 Open Dir → 选择 dataset/images
  2. 点击 Change Save Dir → 选择 dataset/labels
  3. 在右侧选择标注格式：
    • YOLO（推荐）
    • PascalVOC（XML）

方法 2：画框后直接输入新类别

1

第一次绘制标注时弹窗 → 输入 `gamer` 即可。

开始标注图片

  1. 按 W 或点击 Create RectBox 进入画框模式
  2. 鼠标拖动框选目标
  3. 松开后弹出类别选择窗口 → 选择 gamer
  4. 点击 Next Image 切换下一张

保存标注

  • 如果未开启自动保存，每张图片标注完成后需要点击 Save
  • 生成的 txt 文件出现在：dataset/labels/xxx.txt

开启自动保存（强烈推荐） 路径：Menu → View → Auto Save Mode 开启后：

  • 每画一个框
  • 每修改一个框
  • 每切换一张图都会自动保存标注文件，无需再点击 Save。

import os
import shutil
import random

# 原始路径
images_dir = "./dataset/images"
labels_dir = "./dataset/labels"

# 目标路径
train_images = "./dataset/train/images"
train_labels = "./dataset/train/labels"
val_images = "./dataset/val/images"
val_labels = "./dataset/val/labels"

# 创建目录
for path in [train_images, train_labels, val_images, val_labels]:
    os.makedirs(path, exist_ok=True)

# 获取所有图片
images = [f for f in os.listdir(images_dir) if f.endswith(".jpg") or f.endswith(".png")]
images.sort()  # 保持顺序，可选

# 打乱顺序
random.shuffle(images)

# 划分比例
train_ratio = 0.8
train_count = int(len(images) * train_ratio)

train_files = images[:train_count]
val_files = images[train_count:]

# 复制训练集
for f in train_files:
    shutil.copy(os.path.join(images_dir, f), train_images)
    txt_file = f.replace(".jpg", ".txt").replace(".png", ".txt")
    shutil.copy(os.path.join(labels_dir, txt_file), train_labels)

# 复制验证集
for f in val_files:
    shutil.copy(os.path.join(images_dir, f), val_images)
    txt_file = f.replace(".jpg", ".txt").replace(".png", ".txt")
    shutil.copy(os.path.join(labels_dir, txt_file), val_labels)

print(f"训练集: {len(train_files)} 张图片")
print(f"验证集: {len(val_files)} 张图片")

YOLOv8 训练命令训练命令 

1
2
3
4

yolo task=detect mode=train \
model=yolov8s.pt \
data=D:/PyProject/dataset/dataset.yaml \
epochs=50 \
imgsz=640 \
batch=16 \
name=gamer_exp

参数： task=detect ：目标检测 mode=train ：训练模式 model=yolov8s.pt ：使用预训练的 yolov8 small data ：数据集 yaml 文件 epochs ：训练轮数 imgsz ：图片尺寸 batch ：batch 大小 name ：实验名称
