import cv2
import os

video = cv2.VideoCapture("./wendao.mkv")  
num = 0
save_step = 30  # 每隔 30 帧保存一张
save_dir = "./data/shubiao/images/"

os.makedirs(save_dir, exist_ok=True)

while True:
    ret, frame = video.read()
    if not ret:
        break
    num += 1
    if num % save_step == 0:
        cv2.imwrite(f"{save_dir}/{num}.bmp", frame)

video.release()

LabelImg 数据标注流程

打开 LabelImg 并设置保存路径

  1. 点击 Open Dir → 选择 dataset/images
  2. 点击 Change Save Dir → 选择 dataset/labels
  3. 在右侧选择标注格式：
    • YOLO（推荐）
    • PascalVOC（XML）

开始标注图片

  1. 按 W 或点击 Create RectBox 进入画框模式
  2. 鼠标拖动框选目标
  3. 松开后弹出类别选择窗口 → 配置类别
  4. 点击 Next Image 切换下一张

保存标注

  • 如果未开启自动保存，每张图片标注完成后需要点击 Save
  • 生成的 txt 文件出现在：./data/shubiao//labels/xxx.txt

开启自动保存（强烈推荐） 路径：Menu → View → Auto Save Mode 开启后：

  • 每画一个框
  • 每修改一个框
  • 每切换一张图都会自动保存标注文件，无需再点击 Save。

import os
import shutil
import random

# 原始路径
images_dir = "./data/shubiao/images"
labels_dir = "./data/shubiao/labels"

# 目标路径
train_images = "./data/shubiao/train/images"
train_labels = "./data/shubiao/train/labels"
val_images = "./data/shubiao/val/images"
val_labels = "./data/shubiao/val/labels"

# 创建目录
for path in [train_images, train_labels, val_images, val_labels]:
    os.makedirs(path, exist_ok=True)

# 获取所有图片
images = [f for f in os.listdir(images_dir) if f.endswith(".bmp") or f.endswith(".bmp")]
images.sort()  # 保持顺序，可选

# 打乱顺序
random.shuffle(images)

# 划分比例
train_ratio = 0.8
train_count = int(len(images) * train_ratio)

train_files = images[:train_count]
val_files = images[train_count:]

# 复制训练集
for f in train_files:
    shutil.copy(os.path.join(images_dir, f), train_images)
    txt_file = f.replace(".jpg", ".txt").replace(".bmp", ".txt")
    shutil.copy(os.path.join(labels_dir, txt_file), train_labels)

# 复制验证集
for f in val_files:
    shutil.copy(os.path.join(images_dir, f), val_images)
    txt_file = f.replace(".jpg", ".txt").replace(".bmp", ".txt")
    shutil.copy(os.path.join(labels_dir, txt_file), val_labels)

print(f"训练集: {len(train_files)} 张图片")
print(f"验证集: {len(val_files)} 张图片")

from ultralytics import YOLOv10
if __name__ == "__main__":
    model_yaml_path = "ultralytics\\cfg\\models\\v10\\yolov10n.yaml"
    model = YOLOv10(model_yaml_path).load("yolov10n.pt")
    data_path = '.\\data\\shubiao.yaml'
    results = model.train(
            data=data_path,
            imgsz=640,
            epochs=50,
            batch=1,
            workers=0,
            device='0'
            )

参数： task=detect ：目标检测 mode=train ：训练模式 model=yolov8s.pt ：使用预训练的 yolov8 small data ：数据集 yaml 文件 epochs ：训练轮数 batch ：batch 大小

from ultralytics import YOLOv10

# 加载需要导出的模型
model = YOLOv10('best.pt')

# 导出ONNX模型
model.export(
    format='onnx',        # 指定导出格式为ONNX
    imgsz=640,            # 输入图像尺寸 (训练默认640x640)
    opset=17,             # ONNX算子集版本 (推荐12-18)
    simplify=True,        # 启用模型简化 (强烈推荐)
    dynamic=True,         # 允许动态输入尺寸 (可选)
    batch=1,              # 批次大小 (默认1)
    device='cpu',         # 使用CPU导出 (避免CUDA兼容性问题)
    half=False,           # 禁用FP16 (避免某些设备兼容性问题)
    workspace=4,          # TensorRT工作空间大小 (GB), ONNX导出可忽略
)
